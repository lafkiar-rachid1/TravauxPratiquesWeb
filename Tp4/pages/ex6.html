6.1 Système de couleurs

:root {
  --bg: #ffffff;
  --text: #1f2937;
  --primary: #2563eb;
}

[data-theme="dark"] {
  --bg: #111827;
  --text: #f9fafb;
  --primary: #60a5fa;
}

body {
  background: var(--bg);
  color: var(--text);
}
